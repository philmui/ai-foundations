    image_dir: Path,

    self.manifest_path = manifest_path

    self.batch_size = batch_size
    self.target_size = target_size
    self.split = split

    # Load and clean manifest (Pandas + Module 6)

__iter__: The Generator Heart

Memory-efficient batch iteration

def __iter__(self) -> Iterator[Dict[str, any]]:
    """Iterate over batches of multimodal data."""
    # Shuffle indices if enabled
    indices = np.arange(len(self.df))
    if self.shuffle:
        np.random.shuffle(indices)

    # Generate batches using generator pattern (Module 2)
    for start_idx in range(0, len(self.df), self.batch_size):
        end_idx = min(start_idx + self.batch_size, len(self.df))
        batch_indices = indices[start_idx:end_idx]

        # Load images and text for this batch
        batch_df = self.df.iloc[batch_indices]
        images = [self._load_and_preprocess_image(row)
                  for row in batch_df.itertuples()]

        # Stack into NumPy batch (Module 3-4)
        image_batch = np.stack(images, axis=0)

        yield {
            'image_batch': image_batch,
            'text_batch': batch_df['caption'].tolist(),
            'metadata': {...}
        }

Image Processing Pipeline

Step-by-Step

  • Load: cv2.imread(path)
  • Resize: cv2.resize(img, target_size)
  • Color: cv2.cvtColor(img, BGR2RGB)
  • Normalize: img.astype(float32) / 255.0
  • Return: shape (H, W, C)

Output Shape

The batch dictionary structure

{
    "image_batch": np.ndarray,
        shape: (B, H, W, C)
        example: (4, 224, 224, 3)
        dtype: float32
        range: [0.0, 1.0]

    "text_batch": List[str],
        length: B
        example: ["caption 1", "caption 2", ...]

    "metadata": Dict,
        categories: List[str]
        filenames: List[str]
        split: str
}

Connection to PyTorch/TensorFlow

PyTorch

import torch

# Zero-copy conversion
tensor = torch.from_numpy(
    batch['image_batch']
)

# Permute to (B, C, H, W)
tensor = tensor.permute(0, 3, 1, 2)

# Forward pass
output = model(tensor)
  • Zero-copy from NumPy
  • Channel-first format
  • Gradient tracking

TensorFlow

import tensorflow as tf

# Direct conversion
tensor = tf.convert_to_tensor(
    batch['image_batch']
)

# Already (B, H, W, C)
# No permute needed!

# Forward pass
output = model(tensor)
  • Direct conversion
  • Channel-last format
  • Eager execution
